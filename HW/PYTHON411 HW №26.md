```python
def is_bad_review(
    review_text: str,
    api_key: str = MISTRAL_API_KEY,
    grades: dict = MISTRAL_MODERATIONS_GRADES,
) -> bool:
    # –°–æ–∑–¥–∞–µ–º –∫–ª–∏–µ–Ω—Ç–∞ Mistral —Å –ø–µ—Ä–µ–¥–∞–Ω–Ω—ã–º API –∫–ª—é—á–æ–º

    # –§–æ—Ä–º–∏—Ä—É–µ–º –∑–∞–ø—Ä–æ—Å
    response = client.classifiers.moderate_chat(
        model="mistral-moderation-latest",
        inputs=[{"role": "user", "content": review_text}],
    )
    # –í—ã—Ç–∞—Å–∫–∏–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ —Å –æ—Ü–µ–Ω–∫–æ–π
    result = response.results[0].category_scores

    # –û–∫—Ä—É–≥–ª—è–µ–º –∑–Ω–∞—á–µ–Ω–∏—è –¥–æ –¥–≤—É—Ö –∑–Ω–∞–∫–æ–≤ –ø–æ—Å–ª–µ –∑–∞–ø—è—Ç–æ–π
    result = {key: round(value, 2) for key, value in result.items()}

    pprint(result)

    # –°–ª–æ–≤–∞—Ä—å –ø–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø—Ä–æ–≤–µ—Ä–∫–∏

        if key in grades:
            checked_result[key] = value >= grades[key]

    # –ï—Å–ª–∏ –æ–¥–Ω–æ –∏–∑ –∑–Ω–∞—á–µ–Ω–∏–π True, —Ç–æ –æ—Ç–∑—ã–≤ –Ω–µ –ø—Ä–æ—Ö–æ–¥–∏—Ç –º–æ–¥–µ—Ä–∞—Ü–∏—é
    return any(checked_result.values())
```
